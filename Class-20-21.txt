Example 1.1
Constructor with multiple object */

//project Java Constructor 
package org.test;
public class Employee {

	public Employee () {// constructor
		System.out.println("Employee Information");
	}
	public Employee(String name) {
		System.out.println(name);
	}
	public Employee(int id, long phNo) {
		System.out.println("ID is "+ id +  "\nPhone number is "+phNo);
	}
	public Employee(long phNo, int id) {
		System.out.println("Alter phno no is "+phNo + "\nAlter id is "+id);
	}
		public static void main(String[] args) {
    System.out.println("***object created for each constructor***");
		Employee a = new Employee();
		Employee b = new Employee("balaji");
		Employee c = new Employee(10,1234567l);
		Employee d = new Employee(456789756l,12);
		
	}
}
/* runtime output
***object created for each constructor***
Employee Information
balaji
ID is 10
Phone number is 1234567
Alter phno no is 456789756
Alter id is 12
*/





/*Day05-Class20
Example 1.2
Constructor with Single object */

//project Java Constructor 
package org.test;

public class Employee {

	public Employee () {// constructor
		this("balaji");
		System.out.println("Employee Information");//4
	}
	public Employee(String name) {
		this(10,1234567l);
		System.out.println(name);//3
	}
	public Employee(int id, long phNo) {
		this(456789l,12);
		System.out.println("ID is "+ id +  "\nPhone number is "+phNo);//2
	}
	public Employee(long phNo, int id) {
		System.out.println("Alter phno no is "+phNo + "\nAlter id is "+id);//1
	}
		public static void main(String[] args) {
    System.out.println("***single object created for all constructor***");
		Employee a = new Employee();
		
	}
}
/* runtime output***object created for each constructor***
Alter phno no is 456789
Alter id is 12
ID is 10
Phone number is 1234567
balaji
Employee Information
*/





/* Day05-Class21
Example 1.1
Description: project Constructor using constructor chaning this */
//project Java Constructor
package org.test;

public class Company{ //parent
	
	public Company() {//1
		this(100);
		System.out.println("Para construction");//3
	}
	public Company(int id) {//2
		this("tcs");
		System.out.println("Company id is : "+id);//2
	}
	public Company(String name) {//3
		
		System.out.println("Company name is : "+name);//1
	}
    
}

package org.test;
public class Employee extends Company {//child

	public Employee () {//1
		this("baljai");//4
		System.out.println("Employee Information");
	}
	public Employee (String name) {//2
		this(12,12345678l);
		System.out.println("Employee name is : "+name);//3
	}
	public Employee (int iD, long phNo) {//3
		this(234576l,23);
		System.out.println("Employee id and Phone no is : \n	"+iD+phNo);//2
	}
	public Employee (long phNo, int iD) {//4
		
		System.out.println("Employee phno&Id is :\n	"+phNo+iD);//1
	}
	public static void main(String[] args) {
		new Employee();
	}
	
}
/* runtime output
Company name is : tcs
Company id is : 100
Para construction
Employee phno&Id is :
	23457623
Employee id and Phone no is : 
	1212345678
Employee name is : baljai
Employee Information
*/

Description: Constructor using constructor chaining with "Super" Keyword 
project Java Constructor */

package org.test;

public class Company{ //parent
	
	public Company() {//6
		this(34);
		System.out.println("2. Para construction");//2
	}
	public Company(int id) {//7
		System.out.println("1. Company id is : "+id);//1
	}
	public Company(String name) {//5
		this();
		System.out.println("3. Company name is : "+name);//3
	}
    
}

package org.test;
public class Employee extends Company {//child

	public Employee () {//4
		super("TCS");
		System.out.println("4. Employee Information");//4
	}
	public Employee (String name) {//1
		this(10,12345678l);
		System.out.println("7. Employee name is : "+name);//7
	}
	public Employee (int iD, long phNo) {//2
		this(3456782l,21);
		System.out.println("6. Employee id and Phone no is : \n	"+iD+phNo);//6
	}
	public Employee (long phNo, int iD) {//3
		this();
		System.out.println("5. Employee phno&Id is :\n	"+phNo+iD);//5
	}
	public static void main(String[] args) {
		
		Employee e = new Employee("balaji");
	}
	
}
/* runtime output
1. Company id is : 34
2. Para construction
3. Company name is : TCS
4. Employee Information
5. Employee phno&Id is :
	345678221
6. Employee id and Phone no is : 
	1012345678
7. Employee name is : balaji
*/
